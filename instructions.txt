from pathlib import Path
from typing import Any, Dict
import yaml
from pydantic import BaseModel, Field, validator
from pydantic.env_settings import BaseSettings

class KrakenSettings(BaseModel):
    api_key: str = Field(..., min_length=10)
    api_secret: str = Field(..., min_length=10)

class TelegramSettings(BaseModel):
    bot_token: str = Field(..., regex=r"^\d+:[A-Za-z0-9_-]{35}$")
    chat_id: str = Field(..., min_length=1)

class TradingSettings(BaseModel):
    test_mode: bool = True
    default_stake_usd: float = Field(..., gt=0)
    max_daily_loss_usd: float = Field(..., gt=0)
    max_positions: int = Field(..., gt=0, le=20)
    pairs: list[str] = Field(..., min_items=1)

    @validator("pairs")
    def validate_pairs(cls, v):
        # Add Kraken-specific pair validation if needed
        return [p.upper() for p in v]

import time
import hmac

import urllib.parse
from typing import Dict, List, Optional, Any

from utils.logging import get_logger

class KrakenClient:
    """Async client for Kraken API (public + private endpoints)."""
    
    def __init__(self, config: Dict[str, Any]):
        self.logger = get_logger(__name__)
        self.api_key = config.get("api_key")
        self.api_secret = config.get("api_secret")
        self.base_url = "https://api.kraken.com"
        self.session: Optional[aiohttp.ClientSession] = None
        self._nonce_counter = 0
    
    async def start(self) -> None:
        """Initialize HTTP session."""
        self.session = aiohttp.ClientSession()
        self.logger.info("🔌 Kraken client initialized")
    
    async def close(self) -> None:
        """Close HTTP session."""
        if self.session:
            await self.session.close()
    
    def _get_nonce(self) -> str:
        """Generate increasing nonce for API auth."""
        self._nonce_counter += 1
        return str(int(time.time() * 1000) + self._nonce_counter)
    
    def _sign_request(self, urlpath: str, data: Dict[str, Any]) -> str:
        """Create API signature for private endpoints."""
        postdata = urllib.parse.urlencode(data)
        encoded = (str(self._get_nonce()) + postdata).encode()
        message = urlpath.encode() + hashlib.sha256(encoded).digest()
        
        mac = hmac.new(
            base64.b64decode(self.api_secret),
            message,
            hashlib.sha512
        )
        return base64.b64encode(mac.digest()).decode()
    
    async def _private_post(self, endpoint: str, data: Dict[str, Any]) -> Dict[str, Any]:
        """Execute signed POST request to private endpoint."""
        if not self.session:
            raise RuntimeError("Client not started")
        
        headers = {
            "API-Key": self.api_key,
            "API-Sign": self._sign_request(endpoint, data)
        }
        
        url = f"{self.base_url}{endpoint}"
        async with self.session.post(url, headers=headers, data=data) as resp:
            resp.raise_for_status()
            return await resp.json()
    
    async def _public_get(self, endpoint: str, params: Optional[Dict] = None) -> Dict[str, Any]:
        """Execute public GET request."""
        if not self.session:
            raise RuntimeError("Client not started")
        
        url = f"{self.base_url}{endpoint}"
        async with self.session.get(url, params=params) as resp:
            resp.raise_for_status()
            return await resp.json()
